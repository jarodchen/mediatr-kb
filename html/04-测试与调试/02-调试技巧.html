🔍 高效排查 MediatR 应用中的问题

📖 概述 ​

调试 MediatR 应用需要掌握一些特殊技巧，因为请求处理涉及管道行为链、依赖注入和异步执行。本章介绍实用的调试方法和工具。

🛠️ 日志输出请求执行流程 ​

增强版日志行为 ​

csharp
public class DetailedLoggingBehavior<TRequest, TResponse> : IPipelineBehavior<TRequest, TResponse>
    where TRequest : IRequest<TResponse>
{
    private readonly ILogger<DetailedLoggingBehavior<TRequest, TResponse>> _logger;
    private readonly IHttpContextAccessor _httpContextAccessor;

    public DetailedLoggingBehavior(
        ILogger<DetailedLoggingBehavior<TRequest, TResponse>> logger,
        IHttpContextAccessor httpContextAccessor)
    {
        _logger = logger;
        _httpContextAccessor = httpContextAccessor;
    }

    public async Task<TResponse> Handle(
        TRequest request, 
        RequestHandlerDelegate<TResponse> next, 
        CancellationToken cancellationToken)
    {
        var requestId = Guid.NewGuid();
        var requestName = typeof(TRequest).Name;
        var userId = _httpContextAccessor.HttpContext?.User?.FindFirstValue(ClaimTypes.NameIdentifier);
        
        _logger.LogInformation(
            "═══════════════════════════════════════\n" +
            "📨 开始处理请求\n" +
            "   请求ID: {RequestId}\n" +
            "   请求类型: {RequestName}\n" +
            "   用户ID: {UserId}\n" +
            "   时间戳: {Timestamp}\n" +
            "   请求数据: {@Request}\n" +
            "═══════════════════════════════════════",
            requestId,
            requestName,
            userId ?? "Anonymous",
            DateTime.UtcNow,
            request);

        var stopwatch = Stopwatch.StartNew();

        try
        {
            var response = await next();

            stopwatch.Stop();

            _logger.LogInformation(
                "✅ 请求处理成功\n" +
                "   请求ID: {RequestId}\n" +
                "   耗时: {ElapsedMilliseconds}ms\n" +
                "   响应数据: {@Response}",
                requestId,
                stopwatch.ElapsedMilliseconds,
                response);

            return response;
        }
        catch (Exception ex)
        {
            stopwatch.Stop();

            _logger.LogError(
                ex,
                "❌ 请求处理失败\n" +
                "   请求ID: {RequestId}\n" +
                "   耗时: {ElapsedMilliseconds}ms\n" +
                "   异常类型: {ExceptionType}\n" +
                "   异常消息: {Message}",
                requestId,
                stopwatch.ElapsedMilliseconds,
                ex.GetType().Name,
                ex.Message);

            throw;
        }
    }
}

调试行为链 ​

csharp
public class PipelineTraceBehavior<TRequest, TResponse> : IPipelineBehavior<TRequest, TResponse>
    where TRequest : IRequest<TResponse>
{
    private static int _behaviorCounter = 0;
    private readonly int _order;
    private readonly string _behaviorName;

    public PipelineTraceBehavior()
    {
        _order = Interlocked.Increment(ref _behaviorCounter);
        _behaviorName = GetType().Name;
    }

    public async Task<TResponse> Handle(
        TRequest request, 
        RequestHandlerDelegate<TResponse> next, 
        CancellationToken cancellationToken)
    {
        var indent = new string(' ', _order * 2);
        Console.WriteLine($"{indent}➡️  进入 [{_order}] {_behaviorName}");

        try
        {
            var response = await next();
            
            Console.WriteLine($"{indent}⬅️  离开 [{_order}] {_behaviorName}");
            
            return response;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"{indent}❌ 异常 [{_order}] {_behaviorName}: {ex.Message}");
            throw;
        }
    }
}

// 注册（按顺序）
builder.Services.AddTransient(typeof(IPipelineBehavior<,>), typeof(PipelineTraceBehavior<,>)); // Logging
builder.Services.AddTransient(typeof(IPipelineBehavior<,>), typeof(PipelineTraceBehavior<,>)); // Validation
builder.Services.AddTransient(typeof(IPipelineBehavior<,>), typeof(PipelineTraceBehavior<,>)); // Transaction

🔬 诊断监听器与 DiagnosticSource ​

创建诊断监听器 ​

csharp
public class MediatRDiagnosticsObserver : IObserver<KeyValuePair<string, object>>
{
    private readonly ILogger<MediatRDiagnosticsObserver> _logger;
    private readonly ActivitySource _activitySource;

    public MediatRDiagnosticsObserver(
        ILogger<MediatRDiagnosticsObserver> logger,
        ActivitySource activitySource)
    {
        _logger = logger;
        _activitySource = activitySource;
    }

    public void OnNext(KeyValuePair<string, object> kvp)
    {
        switch (kvp.Key)
        {
            case "MediatR.Request.Start":
                HandleRequestStart(kvp.Value);
                break;
            case "MediatR.Request.Stop":
                HandleRequestStop(kvp.Value);
                break;
            case "MediatR.Request.Error":
                HandleRequestError(kvp.Value);
                break;
            case "MediatR.Notification.Start":
                HandleNotificationStart(kvp.Value);
                break;
        }
    }

    private void HandleRequestStart(object data)
    {
        var requestData = (dynamic)data;
        _logger.LogInformation("🚀 请求开始: {RequestName}", requestData.RequestName);
        
        // 创建 Activity 用于分布式追踪
        var activity = _activitySource.StartActivity($"MediatR.{requestData.RequestName}");
        activity?.SetTag("request.name", requestData.RequestName);
    }

    private void HandleRequestStop(object data)
    {
        var responseData = (dynamic)data;
        _logger.LogInformation(
            "✅ 请求完成: {RequestName}, 耗时: {Duration}ms",
            responseData.RequestName,
            responseData.Duration);
    }

    private void HandleRequestError(object data)
    {
        var errorData = (dynamic)data;
        _logger.LogError(
            errorData.Exception,
            "❌ 请求失败: {RequestName}, 错误: {Message}",
            errorData.RequestName,
            errorData.Exception.Message);
    }

    public void OnCompleted() { }
    public void OnError(Exception error) { }
}

[MemoryDiagnoser]

public class MediatRBenchmarks
{
    private IMediator _mediator;
    private CreateOrderCommand _command;

    [GlobalSetup]
    public void Setup()
    {
        var services = new ServiceCollection();
        
        services.AddMediatR(cfg => 
            cfg.RegisterServicesFromAssembly(typeof(CreateOrderHandler).Assembly));
        
        services.AddScoped<IOrderRepository, InMemoryOrderRepository>();
        
        var provider = services.BuildServiceProvider();
        _mediator = provider.GetRequiredService<IMediator>();
        
        _command = new CreateOrderCommand
        {
            ProductName = "Benchmark Product",
            Quantity = 1,
            Price = 100
        };
    }

    [Benchmark]
    public async Task<OrderResult> SendCommand()
    {
        return await _mediator.Send(_command);
    }

    [Benchmark]
    public async Task PublishNotification()
    {
        await _mediator.Publish(new OrderCreatedNotification
        {
            OrderId = Guid.NewGuid(),
            TotalAmount = 100
        });
    }
}

// 运行基准测试
public class Program
{
    public static void Main(string[] args)
    {
        var summary = BenchmarkRunner.Run<MediatRBenchmarks>();
    }
}

🐛 常见问题排查 ​

问题 1：Handler 未被发现 ​

症状：InvalidOperationException: No service for type 'IRequestHandler'

排查步骤：

csharp
// 1. 检查程序集扫描
var assembly = typeof(Program).Assembly;
Console.WriteLine($"扫描程序集: {assembly.FullName}");

// 2. 列出所有注册的 Handler
var handlers = assembly.GetTypes()
    .Where(t => t.GetInterfaces().Any(i => i.IsGenericType && 
           i.GetGenericTypeDefinition() == typeof(IRequestHandler<,>)))
    .ToList();

Console.WriteLine($"找到 {handlers.Count} 个 Handler:");
foreach (var handler in handlers)
{
    Console.WriteLine($"  - {handler.FullName}");
}

// 3. 确认 Handler 是 public 且非抽象
public class MyHandler : IRequestHandler<MyCommand> // ✅ 正确
internal class MyHandler : IRequestHandler<MyCommand> // ❌ 错误：必须是 public

问题 4：通知处理器异常中断 ​

症状：一个处理器失败导致其他处理器不执行

解决方案：使用容错 Mediator

csharp
public class ResilientMediator : Mediator
{
    private readonly ILogger<ResilientMediator> _logger;

    protected override async Task PublishCore(
        IEnumerable<NotificationHandlerExecutor> handlerExecutors, 
        INotification notification, 
        CancellationToken cancellationToken)
    {
        foreach (var handlerExecutor in handlerExecutors)
        {
            try
            {
                await handlerExecutor.HandlerCallback(notification, cancellationToken);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, 
                    "通知处理器失败，但继续执行其他处理器: {HandlerType}",
                    handlerExecutor.HandlerInstance.GetType().Name);
            }
        }
    }
}

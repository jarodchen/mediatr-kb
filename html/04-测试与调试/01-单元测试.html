单元测试 ​

🧪 确保 MediatR 应用的质量和可靠性

📖 概述 ​

单元测试是保证 MediatR 应用质量的关键环节。由于 MediatR 的设计天然支持测试（Handler 是纯函数、依赖注入清晰），编写单元测试相对简单且高效。

测试目标 ​

  • ✅ 验证 Handler 业务逻辑正确性
  • ✅ 测试管道行为的横切关注点
  • ✅ 验证通知处理器的多播机制
  • ✅ 确保异常处理符合预期
  • ✅ Mock IMediator 进行集成测试

🔧 测试环境搭建 ​

安装 NuGet 包 ​

bash
# 创建测试项目
dotnet new xunit -n YourProject.Tests

# 添加引用
cd YourProject.Tests
dotnet add reference ../YourProject/YourProject.csproj

# 安装测试框架和 Mock 库
dotnet add package xunit
dotnet add package xunit.runner.visualstudio
dotnet add package Microsoft.NET.Test.Sdk
dotnet add package Moq
dotnet add package FluentAssertions  # 可选，更优雅的断言

public class GetOrderHandler : IRequestHandler<GetOrderQuery, OrderDto>
{
    private readonly IOrderRepository _repo;
    private readonly IMapper _mapper;

    public GetOrderHandler(IOrderRepository repo, IMapper mapper)
    {
        _repo = repo;
        _mapper = mapper;
    }

    public async Task<OrderDto> Handle(GetOrderQuery request, CancellationToken ct)
    {
        var order = await _repo.GetByIdAsync(request.OrderId, ct);
        
        if (order == null)
            throw new NotFoundException($"Order {request.OrderId} not found");

        return _mapper.Map<OrderDto>(order);
    }
}

// 测试
public class GetOrderHandlerTests
{
    private readonly Mock<IOrderRepository> _mockRepo;
    private readonly Mock<IMapper> _mockMapper;
    private readonly GetOrderHandler _handler;

    public GetOrderHandlerTests()
    {
        _mockRepo = new Mock<IOrderRepository>();
        _mockMapper = new Mock<IMapper>();
        _handler = new GetOrderHandler(_mockRepo.Object, _mockMapper.Object);
    }

    [Fact]
    public async Task Handle_ExistingOrder_ShouldReturnDto()
    {
        // Arrange
        var orderId = Guid.NewGuid();
        var order = new Order { Id = orderId, ProductName = "iPhone" };
        var dto = new OrderDto { Id = orderId, ProductName = "iPhone" };

        _mockRepo.Setup(r => r.GetByIdAsync(orderId, It.IsAny<CancellationToken>()))
            .ReturnsAsync(order);
        
        _mockMapper.Setup(m => m.Map<OrderDto>(order))
            .Returns(dto);

        var query = new GetOrderQuery { OrderId = orderId };

        // Act
        var result = await _handler.Handle(query, CancellationToken.None);

        // Assert
        result.Should().BeEquivalentTo(dto);
        _mockRepo.Verify(r => r.GetByIdAsync(orderId, It.IsAny<CancellationToken>()), Times.Once);
    }

    [Fact]
    public async Task Handle_NonExistentOrder_ShouldThrowNotFoundException()
    {
        // Arrange
        var orderId = Guid.NewGuid();
        
        _mockRepo.Setup(r => r.GetByIdAsync(orderId, It.IsAny<CancellationToken>()))
            .ReturnsAsync((Order)null);

        var query = new GetOrderQuery { OrderId = orderId };

        // Act & Assert
        var exception = await Assert.ThrowsAsync<NotFoundException>(() => 
            _handler.Handle(query, CancellationToken.None));
        
        exception.Message.Should().Contain(orderId.ToString());
    }
}

public class LoggingBehaviorTests
{
    private readonly Mock<ILogger<LoggingBehavior<TestRequest, TestResponse>>> _mockLogger;
    private readonly LoggingBehavior<TestRequest, TestResponse> _behavior;

    public LoggingBehaviorTests()
    {
        _mockLogger = new Mock<ILogger<LoggingBehavior<TestRequest, TestResponse>>>();
        _behavior = new LoggingBehavior<TestRequest, TestResponse>(_mockLogger.Object);
    }

    [Fact]
    public async Task Handle_ShouldLogStartAndEnd()
    {
        // Arrange
        var request = new TestRequest { Data = "test" };
        var response = new TestResponse { Result = "success" };
        
        RequestHandlerDelegate<TestResponse> next = () => Task.FromResult(response);

        // Act
        var result = await _behavior.Handle(request, next, CancellationToken.None);

        // Assert
        result.Should().BeEquivalentTo(response);
        
        // 验证日志被调用
        _mockLogger.Verify(
            x => x.Log(
                LogLevel.Information,
                It.IsAny<EventId>(),
                It.Is<It.IsAnyType>((v, t) => v.ToString().Contains("开始处理请求")),
                null,
                It.IsAny<Func<It.IsAnyType, Exception, string>>()),
            Times.AtLeastOnce);
    }

    [Fact]
    public async Task Handle_Exception_ShouldLogError()
    {
        // Arrange
        var request = new TestRequest();
        
        RequestHandlerDelegate<TestResponse> next = () => throw new InvalidOperationException("Test error");

        // Act & Assert
        await Assert.ThrowsAsync<InvalidOperationException>(() => 
            _behavior.Handle(request, next, CancellationToken.None));

        _mockLogger.Verify(
            x => x.Log(
                LogLevel.Error,
                It.IsAny<EventId>(),
                It.Is<It.IsAnyType>((v, t) => v.ToString().Contains("请求处理失败")),
                It.IsAny<Exception>(),
                It.IsAny<Func<It.IsAnyType, Exception, string>>()),
            Times.Once);
    }
}

public class ValidationBehaviorTests
{
    private readonly Mock<IValidator<TestRequest>> _mockValidator;
    private readonly ValidationBehavior<TestRequest, TestResponse> _behavior;

    public ValidationBehaviorTests()
    {
        _mockValidator = new Mock<IValidator<TestRequest>>();
        _behavior = new ValidationBehavior<TestRequest, TestResponse>(
            new[] { _mockValidator.Object });
    }

    [Fact]
    public async Task Handle_ValidRequest_ShouldCallNext()
    {
        // Arrange
        var request = new TestRequest { Name = "Valid" };
        var response = new TestResponse();
        
        _mockValidator.Setup(v => v.ValidateAsync(It.IsAny<ValidationContext<TestRequest>>(), It.IsAny<CancellationToken>()))
            .ReturnsAsync(new ValidationResult()); // 验证通过

        bool nextCalled = false;
        RequestHandlerDelegate<TestResponse> next = () => 
        {
            nextCalled = true;
            return Task.FromResult(response);
        };

        // Act
        var result = await _behavior.Handle(request, next, CancellationToken.None);

        // Assert
        nextCalled.Should().BeTrue();
        result.Should().Be(response);
    }

    [Fact]
    public async Task Handle_InvalidRequest_ShouldThrowValidationException()
    {
        // Arrange
        var request = new TestRequest { Name = "" }; // 无效数据
        
        var failures = new List<ValidationFailure>
        {
            new ValidationFailure("Name", "Name cannot be empty")
        };
        
        _mockValidator.Setup(v => v.ValidateAsync(It.IsAny<ValidationContext<TestRequest>>(), It.IsAny<CancellationToken>()))
            .ReturnsAsync(new ValidationResult(failures));

        RequestHandlerDelegate<TestResponse> next = () => Task.FromResult(new TestResponse());

        // Act & Assert
        var exception = await Assert.ThrowsAsync<ValidationException>(() => 
            _behavior.Handle(request, next, CancellationToken.None));
        
        exception.Errors.Should().HaveCount(1);
        exception.Errors[0].PropertyName.Should().Be("Name");
    }
}

public class OrderCreatedNotificationTests
{
    private readonly Mock<IEmailService> _mockEmailService;
    private readonly Mock<ICacheService> _mockCacheService;
    private readonly Mock<IAuditService> _mockAuditService;

    public OrderCreatedNotificationTests()
    {
        _mockEmailService = new Mock<IEmailService>();
        _mockCacheService = new Mock<ICacheService>();
        _mockAuditService = new Mock<IAuditService>();
    }

    [Fact]
    public async Task AllHandlers_ShouldBeCalled()
    {
        // Arrange
        var notification = new OrderCreatedNotification
        {
            OrderId = Guid.NewGuid(),
            CustomerEmail = "test@example.com"
        };

        var emailHandler = new SendEmailHandler(_mockEmailService.Object);
        var cacheHandler = new UpdateCacheHandler(_mockCacheService.Object);
        var auditHandler = new AuditLogHandler(_mockAuditService.Object);

        // Act - 并行执行所有处理器
        await Task.WhenAll(
            emailHandler.Handle(notification, CancellationToken.None),
            cacheHandler.Handle(notification, CancellationToken.None),
            auditHandler.Handle(notification, CancellationToken.None)
        );

        // Assert - 验证所有处理器都被调用
        _mockEmailService.Verify(e => e.SendAsync(notification.CustomerEmail, It.IsAny<string>()), Times.Once);
        _mockCacheService.Verify(c => c.InvalidateAsync(It.IsAny<string>()), Times.Once);
        _mockAuditService.Verify(a => a.LogAsync(notification.OrderId), Times.Once);
    }

    [Fact]
    public async Task OneHandlerFails_OthersShouldStillExecute()
    {
        // Arrange
        var notification = new OrderCreatedNotification
        {
            OrderId = Guid.NewGuid(),
            CustomerEmail = "test@example.com"
        };

        _mockEmailService.Setup(e => e.SendAsync(It.IsAny<string>(), It.IsAny<string>()))
            .ThrowsAsync(new InvalidOperationException("Email service down"));

        var emailHandler = new SendEmailHandler(_mockEmailService.Object);
        var cacheHandler = new UpdateCacheHandler(_mockCacheService.Object);

        // Act - 即使一个失败，其他仍应执行
        var emailTask = emailHandler.Handle(notification, CancellationToken.None);
        var cacheTask = cacheHandler.Handle(notification, CancellationToken.None);

        await Assert.ThrowsAsync<InvalidOperationException>(() => emailTask);
        await cacheTask; // 这个应该成功

        // Assert
        _mockCacheService.Verify(c => c.InvalidateAsync(It.IsAny<string>()), Times.Once);
    }
}

        var command = new CreateOrderCommand 
        {

            Quantity = 1

        // Act
        var result = await _controller.GetOrder(orderId);

        // Assert
        result.Result.Should().BeOfType<NotFoundResult>();
    }
}

🎓 总结 ​

核心价值 ​

✅ Handler 易于测试：纯函数、依赖清晰
✅ Mock 简单：标准依赖注入模式
✅ 隔离性好：无需启动完整应用
✅ 快速反馈：单元测试运行速度快

关键要点 ​

  1. 每个 Handler 独立测试
  2. 使用 Mock 隔离外部依赖
  3. 测试正常流程和异常流程
  4. 使用 Theory 进行参数化测试
  5. 集成测试验证端到端流程

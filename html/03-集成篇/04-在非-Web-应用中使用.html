// Commands/ProcessDataCommand.cs
using MediatR;

public record ProcessDataCommand(string FilePath) : IRequest<ProcessResult>;

public record ProcessResult
{
    public bool Success { get; init; }
    public int RecordsProcessed { get; init; }
    public string? ErrorMessage { get; init; }
}

// Handlers/ProcessDataHandler.cs
public class ProcessDataHandler : IRequestHandler<ProcessDataCommand, ProcessResult>
{
    private readonly ILogger<ProcessDataHandler> _logger;
    
    public ProcessDataHandler(ILogger<ProcessDataHandler> logger)
    {
        _logger = logger;
    }
    
    public async Task<ProcessResult> Handle(
        ProcessDataCommand request, 
        CancellationToken cancellationToken)
    {
        try
        {
            _logger.LogInformation("开始处理文件: {FilePath}", request.FilePath);
            
            // 模拟数据处理
            await Task.Delay(1000, cancellationToken);
            
            var result = new ProcessResult
            {
                Success = true,
                RecordsProcessed = 1000
            };
            
            _logger.LogInformation("处理完成: {RecordsProcessed} 条记录", 
                result.RecordsProcessed);
            
            return result;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "处理失败");
            
            return new ProcessResult
            {
                Success = false,
                ErrorMessage = ex.Message
            };
        }
    }
}

// Program.cs
using MediatR;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Logging;

var services = new ServiceCollection();

// 配置日志
services.AddLogging(builder =>

    builder.AddConsole();
    builder.SetMinimumLevel(LogLevel.Information);
});

// 注册 MediatR

    cfg.RegisterServicesFromAssembly(typeof(Program).Assembly);

    Console.WriteLine($"成功处理 {result.RecordsProcessed} 条记录");
}
else
{
    Console.WriteLine($"处理失败: {result.ErrorMessage}");
}

Console.WriteLine("按任意键退出...");

// Behaviors/LoggingBehavior.cs
public class LoggingBehavior<TRequest, TResponse> 
    : IPipelineBehavior<TRequest, TResponse>
    where TRequest : IRequest<TResponse>
{
    private readonly ILogger<LoggingBehavior<TRequest, TResponse>> _logger;
    
    public async Task<TResponse> Handle(
        TRequest request,
        RequestHandlerDelegate<TResponse> next,
        CancellationToken cancellationToken)
    {
        _logger.LogInformation("执行命令: {RequestType}", typeof(TRequest).Name);
        
        var stopwatch = Stopwatch.StartNew();
        
        try
        {
            var response = await next();
            
            stopwatch.Stop();
            _logger.LogInformation("命令执行成功，耗时: {ElapsedMs}ms", 
                stopwatch.ElapsedMilliseconds);
            
            return response;
        }
        catch (Exception ex)
        {
            stopwatch.Stop();
            _logger.LogError(ex, "命令执行失败，耗时: {ElapsedMs}ms", 
                stopwatch.ElapsedMilliseconds);
            throw;
        }
    }
}

// Behaviors/RetryBehavior.cs
public class RetryBehavior<TRequest, TResponse> 
    : IPipelineBehavior<TRequest, TResponse>
    where TRequest : IRequest<TResponse>
{
    private readonly ILogger<RetryBehavior<TRequest, TResponse>> _logger;
    private const int MaxRetries = 3;
    
    public async Task<TResponse> Handle(
        TRequest request,
        RequestHandlerDelegate<TResponse> next,
        CancellationToken cancellationToken)
    {
        Exception? lastException = null;
        
        for (int i = 0; i < MaxRetries; i++)
        {
            try
            {
                return await next();
            }
            catch (Exception ex)
            {
                lastException = ex;
                _logger.LogWarning(ex, 
                    "执行失败，第 {RetryCount} 次重试", i + 1);
                
                if (i < MaxRetries - 1)
                {
                    await Task.Delay(TimeSpan.FromSeconds(Math.Pow(2, i)), 
                        cancellationToken);
                }
            }
        }
        
        throw lastException!;
    }
}

// 注册
services.AddMediatR(cfg =>
{
    cfg.RegisterServicesFromAssembly(typeof(Program).Assembly);
});

services.AddTransient(typeof(IPipelineBehavior<,>), typeof(LoggingBehavior<,>));
services.AddTransient(typeof(IPipelineBehavior<,>), typeof(RetryBehavior<,>));

// Commands/ProcessQueueCommand.cs
public record ProcessQueueCommand : IRequest<Unit>
{
    public string QueueName { get; init; }
    public int BatchSize { get; init; } = 10;
}

// Handlers/ProcessQueueHandler.cs
public class ProcessQueueHandler : IRequestHandler<ProcessQueueCommand, Unit>
{
    private readonly ILogger<ProcessQueueHandler> _logger;
    private readonly IQueueService _queueService;
    
    public ProcessQueueHandler(
        ILogger<ProcessQueueHandler> logger,
        IQueueService queueService)
    {
        _logger = logger;
        _queueService = queueService;
    }
    
    public async Task<Unit> Handle(
        ProcessQueueCommand request, 
        CancellationToken cancellationToken)
    {
        _logger.LogInformation("开始处理队列: {QueueName}, 批次大小: {BatchSize}",
            request.QueueName, 
            request.BatchSize);
        
        var messages = await _queueService.DequeueAsync(
            request.QueueName, 
            request.BatchSize, 
            cancellationToken);
        
        foreach (var message in messages)
        {
            try
            {
                await ProcessMessageAsync(message, cancellationToken);
                await _queueService.AcknowledgeAsync(message.Id, cancellationToken);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "处理消息失败: {MessageId}", message.Id);
                await _queueService.NackAsync(message.Id, cancellationToken);
            }
        }
        
        _logger.LogInformation("队列处理完成: {QueueName}", request.QueueName);
        
        return Unit.Value;
    }
    
    private async Task ProcessMessageAsync(QueueMessage message, CancellationToken ct)
    {
        // 处理消息逻辑
        await Task.Delay(100, ct);
    }
}

// Worker.cs
public class Worker : BackgroundService
{
    private readonly IMediator _mediator;
    private readonly ILogger<Worker> _logger;
    private readonly IConfiguration _configuration;
    
    public Worker(
        IMediator mediator,
        ILogger<Worker> logger,
        IConfiguration configuration)
    {
        _mediator = mediator;
        _logger = logger;
        _configuration = configuration;
    }
    
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("Worker 启动于: {Time}", DateTimeOffset.Now);
        
        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                // 定期处理队列
                await _mediator.Send(new ProcessQueueCommand
                {
                    QueueName = _configuration["QueueName"] ?? "default",
                    BatchSize = 10
                }, stoppingToken);
                
                // 等待一段时间后再次检查
                await Task.Delay(TimeSpan.FromSeconds(5), stoppingToken);
            }
            catch (OperationCanceledException)
            {
                // 正常取消
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Worker 执行出错");
                await Task.Delay(TimeSpan.FromSeconds(10), stoppingToken);
            }
        }
        
        _logger.LogInformation("Worker 停止于: {Time}", DateTimeOffset.Now);
    }
}

// Program.cs
using MediatR;

var host = Host.CreateDefaultBuilder(args)
    .ConfigureServices((hostContext, services) =>
    {
        // 注册 MediatR
        services.AddMediatR(cfg =>
        {
            cfg.RegisterServicesFromAssembly(typeof(Program).Assembly);
        });
        
        // 注册其他服务
        services.AddSingleton<IQueueService, RabbitMqQueueService>();
        
        // 注册 Worker
        services.AddHostedService<Worker>();
    })
    .Build();

await host.RunAsync();

定时任务调度 ​

csharp
// Workers/ScheduledWorker.cs
public class ScheduledWorker : BackgroundService
{
    private readonly IMediator _mediator;
    private readonly ILogger<ScheduledWorker> _logger;
    private Timer? _timer;
    
    public ScheduledWorker(IMediator mediator, ILogger<ScheduledWorker> logger)
    {
        _mediator = mediator;
        _logger = logger;
    }
    
    protected override Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("定时任务 Worker 启动");
        
        // 每天凌晨2点执行
        _timer = new Timer(async _ => await ExecuteScheduledTasks(), 
            null, 
            TimeSpan.Zero, 
            TimeSpan.FromHours(24));
        
        return Task.CompletedTask;
    }
    
    private async Task ExecuteScheduledTasks()
    {
        try
        {
            _logger.LogInformation("执行定时任务");
            
            // 清理过期数据
            await _mediator.Send(new CleanupExpiredDataCommand());
            
            // 生成日报
            await _mediator.Send(new GenerateDailyReportCommand());
            
            // 发送通知
            await _mediator.Send(new SendDailyNotificationCommand());
            
            _logger.LogInformation("定时任务执行完成");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "定时任务执行失败");
        }
    }
    
    public override async Task StopAsync(CancellationToken cancellationToken)
    {
        _logger.LogInformation("定时任务 Worker 停止");
        
        _timer?.Change(Timeout.Infinite, 0);
        
        await base.StopAsync(cancellationToken);
    }
    
    public override void Dispose()
    {
        _timer?.Dispose();
        base.Dispose();
    }
}

// ViewModels/MainViewModel.cs
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using MediatR;

public partial class MainViewModel : ObservableObject
{
    private readonly IMediator _mediator;
    
    [ObservableProperty]
    private ObservableCollection<OrderDto> _orders = new();
    
    [ObservableProperty]
    private bool _isLoading;
    
    [ObservableProperty]
    private string? _statusMessage;
    
    public MainViewModel(IMediator mediator)
    {
        _mediator = mediator;
    }
    
    [RelayCommand]
    private async Task LoadDataAsync()
    {
        if (IsLoading) return;
        
        IsLoading = true;
        StatusMessage = "正在加载数据...";
        
        try
        {
            var result = await _mediator.Send(new LoadOrdersQuery());
            
            Orders.Clear();
            foreach (var order in result)
            {
                Orders.Add(order);
            }
            
            StatusMessage = $"加载完成，共 {result.Count} 条记录";
        }
        catch (Exception ex)
        {
            StatusMessage = $"加载失败: {ex.Message}";
        }
        finally
        {
            IsLoading = false;
        }
    }
    
    [RelayCommand]
    private async Task CreateOrderAsync()
    {
        var dialog = new CreateOrderDialog();
        var result = dialog.ShowDialog();
        
        if (result == true)
        {
            var command = new CreateOrderCommand
            {
                CustomerName = dialog.CustomerName,
                Amount = dialog.Amount
            };
            
            await _mediator.Send(command);
            
            // 重新加载数据
            await LoadDataAsync();
        }
    }
}

// Program.cs
using MediatR;
using Microsoft.Extensions.DependencyInjection;

static class Program
{
    public static IServiceProvider ServiceProvider { get; private set; } = default!;
    
    [STAThread]
    static void Main()
    {
        ApplicationConfiguration.Initialize();
        
        var services = new ServiceCollection();
        
        // 注册 MediatR
        services.AddMediatR(cfg =>
        {
            cfg.RegisterServicesFromAssembly(typeof(Program).Assembly);
        });
        
        // 注册 Form
        services.AddTransient<MainForm>();
        
        ServiceProvider = services.BuildServiceProvider();
        
        Application.Run(ServiceProvider.GetRequiredService<MainForm>());
    }
}

// MainForm.cs
public partial class MainForm : Form
{
    private readonly IMediator _mediator;
    private DataGridView _dataGridView;
    private Button _loadButton;
    
    public MainForm(IMediator mediator)
    {
        _mediator = mediator;
        InitializeComponent();
        SetupUI();

    private void SetupUI()
    {
        _dataGridView = new DataGridView { Dock = DockStyle.Fill };
        _loadButton = new Button { Text = "加载数据", Dock = DockStyle.Top };

        Controls.Add(_dataGridView);

        {
            MessageBox.Show($"加载失败: {ex.Message}");
        }
        finally
        {
            _loadButton.Enabled = true;
        }
    }
}

// Program.cs
using MediatR;

var builder = WebApplication.CreateBuilder(args);

// 添加 Blazor Server 服务
builder.Services.AddRazorPages();
builder.Services.AddServerSideBlazor();

// 注册 MediatR（Scoped 生命周期）
builder.Services.AddMediatR(cfg =>
{
    cfg.RegisterServicesFromAssembly(typeof(Program).Assembly);

注意：在 WebAssembly 中，所有代码都在浏览器中运行，因此：

  • 使用 Singleton 生命周期
  • 需要通过 HTTP API 调用后端
  • Handler 中应该调用 HTTP 客户端而非直接访问数据库
csharp
// Handlers/CreateOrderHandler.cs (WebAssembly)
public class CreateOrderHandler : IRequestHandler<CreateOrderCommand, int>
{
    private readonly HttpClient _httpClient;
    
    public CreateOrderHandler(HttpClient httpClient)
    {
        _httpClient = httpClient;
    }
    
    public async Task<int> Handle(
        CreateOrderCommand request, 
        CancellationToken cancellationToken)
    {
        var response = await _httpClient.PostAsJsonAsync(
            "api/orders", 
            request, 
            cancellationToken);
        
        response.EnsureSuccessStatusCode();
        
        var orderId = await response.Content.ReadFromJsonAsync<int>(cancellationToken);
        
        return orderId;
    }
}
